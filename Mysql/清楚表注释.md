# SQL命令清除注释
mysql清除注释，需要分两步骤！

### 去除表注释SQL
将查询得到SQL语句进行复制，执行即可去除表对应的注释信息。
~~~
SELECT
	concat(
		'alter table ',
		table_schema,
		'.',
		table_name,
		' comment ''',
		''';'
	) table_ant
FROM
	information_schema. COLUMNS
WHERE
	table_schema = '数据库名称'
GROUP BY
	TABLE_NAME;
~~~

### 去除表对应字段注释SQL
将查询得到SQL语句进行复制，执行即可去除表字段对应的注释信息。
~~~
SELECT
	concat(
		'alter table ',
		table_schema,
		'.',
		table_name,
		' modify column `',
		column_name,
		'` ',
		column_type,
	IF (
		is_nullable = 'YES',

	IF (
		data_type IN ('timestamp'),
		' null ',
		' '
	),
	' not null '
	),

IF (
	column_default IS NULL,
	'',

IF (
	data_type IN ('char', 'varchar')
	OR data_type IN ('date', 'datetime')
	AND column_default != 'CURRENT_TIMESTAMP',
	concat(
	' default ''',
		column_default,
		''''
	),
	concat(
		' default ',

	IF (
		column_default = '',
		'''''',
		column_default
	)
	)
)
),

IF (
	extra IS NULL
	OR extra = '',
	'',
	concat(' ', extra)
),
 ' comment ''',
 ''';'
	) s
FROM
	information_schema. COLUMNS
WHERE
	table_schema = '数据库名称';
~~~
